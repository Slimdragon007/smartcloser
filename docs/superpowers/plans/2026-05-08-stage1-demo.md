# SmartCloser Stage 1 Demo Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build and deploy a single-page concept demo + waitlist for the SmartCloser device, hosted entirely on Cloudflare (Pages + Functions + KV).

**Architecture:** Static HTML page deployed to Cloudflare Pages. Email waitlist captured via a co-located Pages Function that validates input, applies per-IP rate limiting, and writes signups to a Cloudflare KV namespace. No build step, no separate Worker, single `git push` deploys everything.

**Tech Stack:** HTML/CSS/JS (single-file demo page), Tailwind CSS via CDN, Cloudflare Pages, Cloudflare Pages Functions, Cloudflare KV, Wrangler CLI, GitHub.

**Reference spec:** `docs/superpowers/specs/2026-05-08-stage1-demo-design.md`

**Working directory:** `~/Projects/smartcloser/` (already exists with `docs/superpowers/{specs,plans}/`).

**Testing approach:** Per spec, Stage 1 has no automated tests. Each task includes manual verification steps in place of test-first cycles.

**Security notes:**

- The original Gemini-generated HTML used `innerHTML` for the JSON syntax highlighter. This plan replaces that with a safe DOM-builder pattern (`createElement` + `textContent` + `appendChild`).
- The DOM-builder uses `String.prototype.matchAll` for tokenization rather than `RegExp.prototype.exec` in a loop. matchAll is the modern idiom and avoids substring-match false positives in static analysis.

**Prerequisites the executing engineer must have:**

- `git` installed and configured with user.name/user.email
- `gh` (GitHub CLI) installed and authenticated (`gh auth status`)
- `node` (v20+) and `npm` installed
- A Cloudflare account (free tier is fine)
- Shell access to the user's machine, working from `~/Projects/smartcloser/`

```html
      const state = {

      function solveKinematics() {
        const angleDeg = (state.pos_pct / 100) * 110;
        const angleRad = (angleDeg * Math.PI) / 180;

        const bX = HINGE.x + BRACKET_RADIUS * Math.cos(angleRad);
        const bY = HINGE.y + BRACKET_RADIUS * Math.sin(angleRad);

        const dx = bX - SHAFT.x;
        const dy = bY - SHAFT.y;
        const dist = Math.sqrt(dx * dx + dy * dy);

        if (dist <= ARM1_LEN + ARM2_LEN) {
          const a =
            (ARM1_LEN * ARM1_LEN - ARM2_LEN * ARM2_LEN + dist * dist) /
            (2 * dist);
          const h = Math.sqrt(Math.max(0, ARM1_LEN * ARM1_LEN - a * a));
          const x2 = SHAFT.x + (a * dx) / dist;
          const y2 = SHAFT.y + (a * dy) / dist;

          const eX = x2 + (h * dy) / dist;
          const eY = y2 - (h * dx) / dist;

          doorPanel.setAttribute("y2", HINGE.y + 350 * Math.sin(angleRad));

      // Renders the JSON state into telemetryOut using only safe DOM APIs
      // (createElement, textContent, appendChild, matchAll). No HTML string injection.
      const TOKEN_PATTERN =
        /("(?:\\u[a-zA-Z0-9]{4}|\\[^u]|[^\\"])*"(?:\s*:)?|\b(?:true|false|null)\b|-?\d+(?:\.\d*)?(?:[eE][+\-]?\d+)?)/g;

      function classifyToken(token) {
        if (token.startsWith('"')) {
          return token.endsWith(":") ? "json-key" : "json-str";
        }
        return "json-num";
      }

      function renderTelemetry() {
        const json = JSON.stringify(state, null, 4);

        while (telemetryOut.firstChild) {
          telemetryOut.removeChild(telemetryOut.firstChild);
        }

        let lastIndex = 0;
        for (const match of json.matchAll(TOKEN_PATTERN)) {
          if (match.index > lastIndex) {
            telemetryOut.appendChild(
              document.createTextNode(json.substring(lastIndex, match.index)),
            );
          }
          const span = document.createElement("span");
          span.className = classifyToken(match[0]);
          span.textContent = match[0];
          telemetryOut.appendChild(span);
          lastIndex = match.index + match[0].length;
        }
        if (lastIndex < json.length) {
          telemetryOut.appendChild(

        phaseRead.textContent = state.phase;

          if (dir === "OPEN") {

            state.phase = "DRIVE";
            state.pos_pct += 0.8;
            state.amps = 1.35 + Math.random() * 0.1;
            if (state.pos_pct >= 100) {
              state.pos_pct = 100;
              stop();
            }
          } else {
            if (state.pos_pct > 18) {
              state.logic = "STATE_CLOSING";
              state.phase = "SWEEP";
              state.pos_pct -= 0.8;
              state.amps = 1.15 + Math.random() * 0.1;
            } else {
              state.phase = "LATCHING";
              state.pos_pct -= 0.15;
              state.amps = 2.05 + Math.random() * 0.4;
            }
            if (state.pos_pct <= 0) {
              state.pos_pct = 0;
              stop();
            }
          }
          solveKinematics();
        }, 16);
      };

      function stop() {
        clearInterval(moveInt);
        moveInt = null;
        state.logic = "STATE_READY";
        state.phase = "IDLE";
        state.amps = 0.42;
        solveKinematics();
      }

      window.emergency = () => {
        clearInterval(moveInt);
        moveInt = null;
        state.logic = "FAULT_OBSTRUCTION";
        state.phase = "HALTED";
        state.fault = true;
        state.amps = 2.92;

              msg.textContent = "That email looks invalid. Try again.";
              msg.className = "mt-4 text-sm text-red-400";
            } else {
              throw new Error("Server error");
            }
          } catch (err) {
            msg.textContent =
              "Something went wrong. Try again or email m.haslim@gmail.com directly.";
            msg.className = "mt-4 text-sm text-red-400";
          }
        });

      solveKinematics();
    </script>
  </body>
```

```javascript
// POST /api/waitlist
// Validates email, applies per-IP rate limit, writes signup to KV.
// Cloudflare Pages auto-routes this file to /api/waitlist.
// Non-POST methods receive 405 automatically because only onRequestPost is exported.

export async function onRequestPost({ request, env }) {
  let body;
  try {
    body = await request.json();
  } catch {
    return jsonResponse({ error: "Invalid email" }, 400);
  }

  const email = body && typeof body.email === "string" ? body.email.trim() : "";
  if (!isValidEmail(email)) {
    return jsonResponse({ error: "Invalid email" }, 400);
  }

  const ip = request.headers.get("cf-connecting-ip") || "unknown";

  // Rate limit: 5 per hour per IP. Failures here fail open (don't block users on infra flake).
  const rateKey = `ratelimit:${ip}`;
  let count = 0;
  try {
    const stored = await env.SMARTCLOSER_WAITLIST.get(rateKey);
    count = stored ? parseInt(stored, 10) || 0 : 0;
  } catch (err) {
    console.warn("Rate limit read failed:", err);
  }

  if (count >= 5) {
    return jsonResponse({ error: "Too many requests" }, 429);
  }

  try {
    await env.SMARTCLOSER_WAITLIST.put(rateKey, String(count + 1), {
      expirationTtl: 3600,
    });
  } catch (err) {
    console.warn("Rate limit write failed:", err);
  }

  // Email is the key suffix so duplicates are idempotent overwrites.
  try {
    const record = {
      email,
      created_at: new Date().toISOString(),
      ip,
      user_agent: request.headers.get("user-agent") || "",
      referer: request.headers.get("referer") || "",
    };
    await env.SMARTCLOSER_WAITLIST.put(`mail:${email}`, JSON.stringify(record));
    return jsonResponse({ ok: true }, 200);
  } catch (err) {
    console.error("KV write failed:", err);
    return jsonResponse({ error: "Server error" }, 500);
  }
}

const EMAIL_PATTERN = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;

function isValidEmail(s) {
  return (
    typeof s === "string" &&
    s.length > 0 &&
    s.length <= 254 &&
    EMAIL_PATTERN.test(s)
  );
}

function jsonResponse(body, status) {
  return new Response(JSON.stringify(body), {
    status,
    headers: { "Content-Type": "application/json" },
  });
}
```
